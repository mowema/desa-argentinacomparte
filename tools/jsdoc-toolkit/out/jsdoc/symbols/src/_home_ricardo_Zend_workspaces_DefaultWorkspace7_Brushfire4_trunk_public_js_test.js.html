1 /**
  2  * Namespace
  3  */
  4 var ZendMax = ZendMax || {};
  5 /**
  6  * Form namespace
  7  */
  8 ZendMax.Form = ZendMax.Form || {}
  9 /**
 10  * Validation function
 11  */
 12 ZendMax.Form.validate = function(form, options)
 13 {
 14     this.submitIfValid =  true;
 15     var settings = $.extend(
 16         {
 17             invalid: function(element, errorMessages) {
 18                 var errorFieldName = 'errorField' + element.attr('name');
 19                 
 20                 // gather all error messages
 21                 var errorMessagesHolder = $('<ul>').addClass(errorFieldName);
 22                 for (var i in errorMessages) {
 23                     var errorMessage = $('<li>').addClass(errorFieldName).html(errorMessages[i]);
 24                     errorMessagesHolder.append(errorMessage);
 25                 }
 26                 
 27                 // decorate the error message
 28                 errorMessagesHolder.css({
 29                     'backgroundColor': '#F00',
 30                     'display': 'block',
 31                     'color': '#FFF',
 32                     'padding': '3px 10px',
 33                     'border-radius': '10px',
 34                     '-moz-border-radius': '10px',
 35                     '-webkit-border-radius': '10px',
 36                     'left': '130px',
 37                     'opacity': '.5',
 38                     'position': 'absolute',
 39                     'top': '0px',
 40                     'z-index': element.attr('z-index')+1
 41                 });
 42                 
 43                 // clear previous error messages for the element
 44                 $('.' + errorFieldName).remove();
 45                 
 46                 // add error messages
 47                 element.addClass('invalid').parent().append(errorMessagesHolder);
 48             }, 
 49             valid: function(element) {
 50                 var errorFieldName = 'errorField' + element.attr('name');
 51                 
 52                 // clear previous error messages for the element
 53                 element.removeClass('invalid');
 54                 $('.' + errorFieldName).remove();
 55             }
 56         },
 57         options
 58     );
 59     
 60     var formId = (form.attr('id')) ? form.attr('id') : 'form';
 61     
 62     // ZendMax.Forms object is created within the form and injected in inlineScript helper and holds all the validation rules and parameters
 63     var formValidates = true;
 64     
 65     for(var elementId in ZendMax.Forms[formId]) {
 66         var element = $("#"+elementId, form);
 67         var elementRuleSet = ZendMax.Forms[formId][elementId];
 68         
 69         var errorMessages = new Array();
 70         for(var i in elementRuleSet) {
 71             
 72             var validatorName = elementRuleSet[i].name;
 73             
 74             // because of the way that Zend_Form handles the isRequired, we must break the validation chain if is not required, and we have no value
 75             // and mark the element as validated
 76             if (validatorName === 'NotRequired' && element.val() === '') {
 77                 break; // break current validation
 78             }
 79             
 80             if (!ZendMax.Form.Validator[validatorName]) {
 81                 console.log("Warning: No Js Validator equivalent was found for " + validatorName +", skipping validation rule");
 82                 continue;
 83             }
 84             
 85             var result = ZendMax.Form.Validator[validatorName](element.val(), elementRuleSet[i].parameters);
 86             if(true !== result.isValid) {
 87                 errorMessages.push(result.errorMessage);
 88                 formValidates = false;
 89                 console.dir(elementRuleSet[0].name);
 90                 console.dir(elementRuleSet[0].parameters);
 91                 // break chain
 92                 if (elementRuleSet[0].name == 'NotEmpty' && elementRuleSet[0].parameters.breakChainOnFailure) {
 93                     break;
 94                 }
 95                 
 96                 if (elementRuleSet[i].parameters.breakChainOnFailure) {
 97                     break;
 98                 }
 99             }
100         }
101         if (errorMessages.length === 0) {
102             settings.valid($(element));
103         } else {
104             settings.invalid($(element), errorMessages);
105         }
106     }
107     
108     if (formValidates) {
109         if(settings.onBeforeSubmit) {
110             settings.onBeforeSubmit(this);
111         }
112     }
113     return formValidates;
114 }
115 /**
116  * Validator class
117  */
118 ZendMax.Form.Validator = {
119     
120     // Helper functions
121     
122     /**
123      * Retrieves the error messages from the validator params
124      * @return string formatted error message
125      */
126     _prepareErrorMessages : function(errorMessage, value, params) {
127         var messageVariables = this._prepareMessageVariables(params);
128         if (null == errorMessage) {
129             errorMessage = 'invalid field';
130             console.log("Warning: No message was received, so 'invalid field' was placed insted");
131         }
132         for (var i in messageVariables) {
133             errorMessage = errorMessage.replace('%'+i+'%', messageVariables[i]);
134         }
135         errorMessage = errorMessage.replace('%value%', value);
136         return errorMessage;
137     },
138     
139     /**
140      * Replaces place holders for real values
141      *   ie: '%min% is less than N caracters long
142      *   will be converted to '3' is less 6 caracters long
143      * @param params object Object with message variables properties.
144      *   ie: max: 50, min: 20
145      * @return array Error messages variables
146      */
147     _prepareMessageVariables: function(params) {
148         var messageVariables = [];
149         for(var i in params.messageVariables) {
150             messageVariables[params.messageVariables[i]] = params[params.messageVariables[i]];
151         }
152         return messageVariables;
153     },
154     
155     // Concrete Validators
156     NotEmpty: function(value, params) {
157         var _return = {isValid: true};
158         
159         if (value == '') {
160             _return.isValid = false;
161             _return.errorMessage = params.messageTemplates.isEmpty;
162         }
163         return _return;
164     },
165     
166     EmailAddress: function(value, params) {
167         var _return = {isValid: true};
168         
169         var emailRegex = /^([\w\!\#$\%\&\'\*\+\-\/\=\?\^\`{\|\}\~]+\.)*[\w\!\#$\%\&\'\*\+\-\/\=\?\^\`{\|\}\~]+@((((([a-z0-9]{1}[a-z0-9\-]{0,62}[a-z0-9]{1})|[a-z])\.)+[a-z]{2,6})|(\d{1,3}\.){3}\d{1,3}(\:\d{1,5})?)$/i;
170         if (emailRegex.test(value)) {
171             return _return;
172         }
173         _return.isValid = false;
174         _return.errorMessage = params.messageTemplates.emailAddressInvalid;
175         return _return;
176     },
177     
178     StringLength: function(value, params) {
179         var _return = {isValid: true};
180         var errorMessage = null;
181         
182         if (value.length < params.min) {
183             _return.isValid = false;
184             errorMessage = params.messageTemplates.stringLengthTooShort;
185         }
186         if (value.length > params.max) {
187             _return.isValid = false;
188             errorMessage = params.messageTemplates.stringLengthTooLong;
189         }
190         if (_return.isValid === false) {
191             _return.errorMessage = this._prepareErrorMessages(
192                 errorMessage,
193                 value,
194                 params
195             );
196         }
197         return _return;
198     }
199 };
